async function rechercherRecette() {
  const nom = document.getElementById("nomRecette").value;
  try {
    const response = await axios.get(
      `https://www.themealdb.com/api/json/v1/1/search.php?s=${nom}`
    );
    afficherRecettes(response.data.meals);
  } catch (error) {
    console.error("Erreur lors de la recherche :", error);
  }
}

function afficherRecettes(recettes) {
  const conteneur = document.getElementById("resultats");
  conteneur.innerHTML = "";
  recettes.forEach((recette) => {
    let div = document.createElement("div");
    div.innerHTML = `<h3>${recette.strMeal}</h3><img src="${recette.strMealThumb}" width="100">`;
    conteneur.appendChild(div);
  });
}

document
  .getElementById("rechercher")
     .addEventListener("click", rechercherRecette);
  
     async function rechercherRecette() {
       const nom = document.getElementById("nomRecette").value;
       try {
         const response = await fetch(
           `https://www.themealdb.com/api/json/v1/1/search.php?s=${nom}`
         );
         const data = await response.json();
         afficherRecettes(data.meals);
       } catch (error) {
         console.error("Erreur lors de la recherche :", error);
       }
     }

     function afficherRecettes(recettes) {
       const conteneur = document.getElementById("resultats");
       conteneur.innerHTML = "";
       if (!recettes) {
         conteneur.innerHTML = "<p>Aucune recette trouv√©e.</p>";
         return;
       }

       recettes.forEach((recette) => {
         let div = document.createElement("div");
         div.classList.add("col-md-4", "mb-4");
         div.innerHTML = `
            <div class="card">
                <img src="${recette.strMealThumb}" class="card-img-top">
                <div class="card-body">
                    <h5 class="card-title">${recette.strMeal}</h5>
                    <a href="${recette.strSource}" target="_blank" class="btn btn-info">Voir la recette</a>
                </div>
            </div>
        `;
         conteneur.appendChild(div);
       });
     }

     document
       .getElementById("rechercher")
       .addEventListener("click", rechercherRecette);
